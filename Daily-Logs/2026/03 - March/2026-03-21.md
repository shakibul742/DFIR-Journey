#### File Analysis
- cat ~/.bashrc
- ls -la ~/.bashrc

#### /tmp & /var/tmp Investigation
- ls -lart /tmp
- ls -lart /var/tmp
- find /tmp -name '*.php'
- find /var/tmp -name '*.sh'
- grep -R "<keyword>" /tmp
- find /tmp -mtime -2
- find /tmp -size +10M
- find /tmp -type f -perm -o=w
